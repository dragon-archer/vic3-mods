on_monthly_pulse_country = {
	events = {
		da.200
	}
	on_actions = {
		da_check_reload
		da_misc_monthly
	}
}

on_yearly_pulse_state = {
	on_actions = {
		da_atheism_spread
	}
}

on_law_enactment_pass = {
	on_actions = {
		da_on_law_enactment_pass
	}
}

on_law_enactment_fail = {
	on_actions = {
		da_on_law_enactment_fail
	}
}

on_law_enactment_ended = {
	on_actions = {
		da_on_law_enactment_ended
	}
}

on_country_formed = {
	on_actions = {
		da_on_country_formed
	}
}

da_on_law_enactment_pass = {
	effect = {
		if = {
			limit = {
				is_subject = yes
				overlord = { is_player = yes }
			}
			da_post_notification = { _ = da_law_pass }
		}
	}
}

da_on_law_enactment_fail = {
	effect = {
		if = {
			limit = {
				is_subject = yes
				overlord = { is_player = yes }
			}
			da_post_notification = { _ = da_law_fail }
		}
	}
}

da_on_law_enactment_ended = {
	effect = {
		if = {
			limit = {
				is_subject = yes
				overlord = { is_player = yes }
			}
			da_post_notification = { _ = da_law_ended }
		}
	}
}

da_on_country_formed = {
	effect = {
		if = {
			limit = {
				exists = c:HRE
				c:HRE ?= ROOT
			}
			da_merge_hre_cultures = yes
		}
		else_if = {
			limit = {
				exists = c:GER
				c:GER ?= ROOT
			}
			da_merge_german_cultures = yes
		}
		else_if = {
			limit = {
				exists = c:ITA
				c:ITA ?= ROOT
			}
			da_merge_italian_cultures = yes
		}
	}
}

da_check_reload = {
	effect = {
		if = {
			limit = {
				is_player = yes
				da_newly_reloaded = yes
			}
			da_reload_effect = yes
		}
	}
}

da_misc_monthly = {
	effect = {
		da_restore_historical_capital = yes
	}
}

da_atheism_spread = {
	effect = {
		random_list = {
			10 = { }
			1 = {
				trigger_event = { id = da_atheism.100 } # Atheist spread
			}
		}
	}
}
