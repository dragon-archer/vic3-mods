
hre_republic = {
	color = "european_blue"

	possible = {
		exists = c:HRE
		THIS = c:HRE
		OR = {
			has_law = law_type:law_parliamentary_republic
			has_law = law_type:law_presidential_republic
			has_law = law_type:law_monarchy
		}
		NOT = {
			has_law = law_type:law_technocracy
		}
		NOT = {
			coa_fascist_trigger = yes
		}
	}
}

hre_technate = {
	color = "technate_grey"

	possible = {
		exists = c:HRE
		THIS = c:HRE
		OR = {
			has_law = law_type:law_parliamentary_republic
			has_law = law_type:law_presidential_republic
		}
		has_law = law_type:law_technocracy
	}
}
