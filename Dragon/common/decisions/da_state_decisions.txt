da_return_treaty_port = {
	is_shown = {
		always = no #TODO
		OR = {
			any_scope_state = {
				is_treaty_port = yes
				state_region = {
					NOT = { has_variable = da_keep_split_var }
					any_scope_state = { owner = { is_subject_of = root } }
				}
			}
			any_subject_or_below = {
				save_temporary_scope_as = o
				any_scope_state = {
					is_treaty_port = yes
					state_region = {
						NOT = { has_variable = da_keep_split_var }
						any_scope_state = {
							OR = {
								owner = root
								owner = {
									is_subject_of = root
									NOT = { this = scope:o }
								}
							}
						}
					}
				}
			}
		}
	}

	possible = {
		custom_tooltip = da_has_treaty_port_of_subject_tt
	}

	when_taken = {
		every_subject_or_below = {
			save_temporary_scope_as = o
			every_scope_state = {
				limit = {
					is_treaty_port = yes
					state_region = {
						NOT = { has_variable = da_keep_split_var }
						any_scope_state = {
							OR = {
								owner = root
								owner = {
									is_subject_of = root
									NOT = { this = scope:o }
								}
							}
						}
					}
				}
				state_region = {
					random_scope_state = {
						limit = {
							OR = {
								owner = root
								owner = {
									is_subject_of = root
									NOT = { this = scope:o }
								}
							}
						}
						owner = {
							save_temporary_scope_as = t
						}
					}
				}
				set_state_owner = scope:t
				save_scope_as = last_state
			}
		}
		every_scope_state = {
			limit = {
				is_treaty_port = yes
				state_region = {
					NOT = { has_variable = da_keep_split_var }
					any_scope_state = {
						OR = {
							owner = root
							owner = { is_subject_of = root }
						}
						count = all
					}
					any_scope_state = { owner = { is_subject_of = root } }
				}
			}
			state_region = {
				random_scope_state = {
					limit = { owner = { is_subject_of = root } }
					owner = { save_temporary_scope_as = t }
				}
			}
			set_state_owner = scope:t
			save_scope_as = last_state #TODO: Update notification
		}
		da_post_notification = { _ = da_decision_return_treaty_port }
	}

	ai_chance = {
		base = 100
	}
}

da_merge_split_state = {
	is_shown = {
		is_player = yes # Temprorarily disable for AI
		any_subject_or_below = {
			any_scope_state = {
				is_capital = no
				save_temporary_scope_as = _
				state_region = {
					NOT = { has_variable = da_keep_split_var }
					any_scope_state = {
						OR = {
							owner = root
							AND = {
								owner = { is_subject_of = root }
								is_largest_state_in_region = yes
								NOT = { this = scope:_ }
							}
						}
					}
				}
			}
		}
	}

	possible = {
		custom_tooltip = da_merge_split_state_tt
	}

	when_taken = {
		every_subject_or_below = {
			every_scope_state = {
				limit = {
					is_capital = no
					state_region = {
						NOT = { has_variable = da_keep_split_var }
						any_scope_state = { owner = root }
					}
				}
				set_state_owner = root
				save_scope_as = last_state
			}
			every_scope_state = {
				limit = {
					is_capital = no
					is_largest_state_in_region = no
					state_region = {
						NOT = { has_variable = da_keep_split_var }
						any_scope_state = { owner = { is_subject_of = root } }
					}
				}
				state_region = {
					every_scope_state = {
						limit = {
							is_largest_state_in_region = yes
							owner = { is_subject_of = root }
						}
						owner = { save_temporary_scope_as = _ }
					}
				}
				if = {
					limit = { exists = scope:_ }
					set_state_owner = scope:_
					save_scope_as = last_state
					clear_saved_scope = _
				}
			}
		}
		da_post_notification = { _ = da_decision_merge_split_state }
	}

	ai_chance = {
		base = 100
	}
}
